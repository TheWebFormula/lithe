Build

Build app with bundles optimized based on routes

Build app

No need for webpack or other bundlers. ESbuild is packed in and pre configured
  • Minification
  • Sourcemaps
  • Dev server
  • live relaoding
  • File copying
  • ENV vars
  • Compression
  • CSP html meta tag (Content-Security-Policy)
  • NODE_ENV=production Automatically adjusts build settings for production
  • NODE_ENV=other_value Automatically adjusts build settings for development
${`import build from '@thewebformula/lithe/build';

/**
* If using 'app/' as root folder then no config needed
*/
build();

/**
* Common usage
*/
build({
  entryPoint: 'docs/app.js',
  entryPointCSS: 'docs/app.css',
  copy: [
    { from: 'docs/_headers', to: 'dist/' },
    { from: 'docs/robots.txt', to: 'dist/' },
    { from: 'docs/sitemap.xml', to: 'dist/' },
    { from: 'docs/favicon.ico', to: 'dist/' }
  ]
});


/**
* Full config options
*/
build({
  // App javascript file : Default 'app/app.js'
  entryPoint: 'app/app.js',

  /**
   * App styles file
   * If you add a path, the file will be bundled and output as a single .css file
   * Default: undefined
   */
  entryPointCSS: 'app/app.css',

  // index.html file : Default '[entryPoint root folder]/index.html'
  indexHTML: 'app/index.html',

  // folder that contains 'app.js' : Default 'dist/'
  outdir: 'dist/',

  /**
  * Minify code
  * Set to 'true' when 'NODE_ENV=production'
  *   otherwise it defaults to 'false'
  * Recommend: do not set
  */
  minify: true,

  /**
  * Create source maps
  * Set to 'false' when 'NODE_ENV=production'
  *   otherwise it defaults to 'true'
  * * Recommend: do not set
  */
  sourcemap: false,

  /**
  * Compress files with Brotli or Gzip
  * Defaults to false
  * If compression is enabled, the dev server will serve the compressed files
  */
  compression: false,
  compressionConfig: {
    type: 'brd',
    level: 19
  },

  // inject env vars to build
  define: {
    'process.env.SOME_VAR': process.env.SOME_VAR
  },

  // supports regex's with wildcards (*, **)
  copy: [
    {
      from: 'app/image.jpg',
      to: 'dist/'
    },
    {
      from: 'app/routes/**/(?!page)*.html',
      to: 'dist/routes'
    }
  ],

  /**
  * Run dev server
  * Set to 'false' when 'NODE_ENV=production'
  * otherwise it defaults to 'true'
  * Recommend: do not set unless you need to change ports
  */
  devServer: {
    /* We recommend not setting enable
     * If not set then it will be controlled by the NODE_ENV variable
     *   NODE_ENV=production will not run the dev server
     *   NODE_ENV=other value will run the dev server
     * if this is set then it will run even if NODE_ENV=production
     */
    enable: true,

    livereload: true,
    host: 'localhost',
    port: 3000,
    cacheHeaders: false,
    
    // SSL options for local https
    keyfile: '/tmp/ssl/dev.local.key',
    certfile: '/tmp/ssl/dev.local.crt',

    cors: {
      origin: 'https://localhost'
    }
  },

  /**
  * Content security policy settings
  * This will inject a meta tag into the index.html file with the specified settings
  * defaults to 'false'
  */
  csp: {
    enable: false,
    requireTrustedTypes: false,
    // There is a built in li-html policy that will auto inject
    trustedTypes: [],
    styleSrc: [self, "https://fonts.googleapis.com"],
    fontSrc: [self, "https://fonts.gstatic.com"]
  },

  /**
  * devWarnings
  * Enable console warning
  * only html sanitization currently
  * defaults to 'false'
  */
  devWarnings: false,

  /**
  * securityLevel
  * Set security level warnings for html
  * Default: 1
  */
  securityLevel: 1,

  // callback before bundle
  onStart: () => {},

  // callback after bundle
  onEnd: () => {}
});`}
